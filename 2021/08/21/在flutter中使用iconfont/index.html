在flutter中使用iconfont

# 安装iconfont_builder

pub global activate iconfont_builder // 全局安装

# 下载iconfont文件

在iconfont中下载图标文件,并放到 assets 文件下

# 修改pubspec.yaml,

  # 图标的字体文件
  fonts:
    - family: Iconfont
      fonts:
        - asset: assets/fonts/iconfont.ttf
          style: italic

# 执行 iconfont_builder的命令

iconfont_builder --from  assets/fonts --to ./lib/untils/Iconfont.dart

# 使用

// 需要导入的生成的Iconfont.dart 文件
Container(
  child: Icon(Iconfont.diqufucengtanchujinggao),
),
